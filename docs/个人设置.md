    <script>
        document.addEventListener('DOMContentLoaded', function() {
            
            // 1. 头像上传预览
            const avatarInput = document.getElementById('avatarInput');
            const avatarPreview = document.getElementById('avatarPreview');
            
            avatarInput.addEventListener('change', function(e) {
                const file = e.target.files[0];
                if (file) {
                    const reader = new FileReader();
                    reader.onload = function(e) {
                        avatarPreview.src = e.target.result;
                        triggerSecurityScan(); // 触发创新点动画
                    }
                    reader.readAsDataURL(file);
                }
            });

            // 2. 密码修改区域展开/收起
            const togglePwdBtn = document.getElementById('togglePasswordBtn');
            const pwdArea = document.getElementById('passwordArea');
            const cancelPwdBtn = document.getElementById('cancelPasswordBtn');

            togglePwdBtn.addEventListener('click', function() {
                pwdArea.classList.add('show');
                triggerSecurityScan(); // 触发创新点动画
            });

            cancelPwdBtn.addEventListener('click', function() {
                pwdArea.classList.remove('show');
            });

            // 3. 保存资料模拟
            const saveBtn = document.getElementById('saveProfileBtn');
            saveBtn.addEventListener('click', function() {
                const originalText = this.textContent;
                this.textContent = '保存中...';
                this.disabled = true;
                this.style.opacity = '0.7';
                
                triggerSecurityScan(); // 触发创新点动画

                setTimeout(() => {
                    this.textContent = '已保存';
                    this.style.background = '#238636'; // 绿色
                    setTimeout(() => {
                        this.textContent = originalText;
                        this.disabled = false;
                        this.style.opacity = '1';
                        this.style.background = ''; // 恢复紫色
                    }, 2000);
                }, 1000);
            });

            // 创新点逻辑：安全扫描动画触发器
            function triggerSecurityScan() {
                const ring = document.getElementById('scanRing');
                ring.classList.add('active');
                ring.style.animationDuration = '0.5s'; // 加速旋转
                
                // 模拟扫描完成
                setTimeout(() => {
                    ring.classList.remove('active');
                    ring.style.animationDuration = '2s'; // 恢复慢速
                }, 2000);
            }
        });
    </script>
