    <script>
        document.addEventListener('DOMContentLoaded', function() {
            // 1. Tab 切换逻辑
            const tabs = document.querySelectorAll('.tab-btn');
            const forms = {
                'password-login': document.getElementById('password-login'),
                'code-login': document.getElementById('code-login')
            };

            tabs.forEach(tab => {
                tab.addEventListener('click', function() {
                    // 移除所有激活状态
                    tabs.forEach(t => t.classList.remove('active'));
                    // 激活当前
                    this.classList.add('active');
                    
                    // 切换表单
                    const targetId = this.getAttribute('data-target');
                    Object.keys(forms).forEach(key => {
                        if (key === targetId) {
                            forms[key].classList.remove('hidden');
                        } else {
                            forms[key].classList.add('hidden');
                        }
                    });
                });
            });

            // 2. 创新点交互：输入框聚焦时加速背景波纹
            const inputs = document.querySelectorAll('.input-field');
            const rings = document.querySelectorAll('.ring');
            
            inputs.forEach(input => {
                input.addEventListener('focus', () => {
                    rings.forEach(ring => {
                        ring.style.animationDuration = '1.5s'; // 加速
                        ring.style.borderColor = '#fff'; // 变白
                    });
                });
                input.addEventListener('blur', () => {
                    rings.forEach(ring => {
                        ring.style.animationDuration = '3s'; // 恢复
                        ring.style.borderColor = 'var(--primary-purple)'; // 恢复紫色
                    });
                });
            });

            // 3. 验证码倒计时模拟
            const sendCodeBtn = document.getElementById('sendCodeBtn');
            sendCodeBtn.addEventListener('click', function() {
                if (this.disabled) return;
                
                let count = 60;
                this.disabled = true;
                this.textContent = `${count}s 后重发`;
                this.style.opacity = '0.5';
                this.style.cursor = 'not-allowed';

                const timer = setInterval(() => {
                    count--;
                    this.textContent = `${count}s 后重发`;
                    if (count <= 0) {
                        clearInterval(timer);
                        this.disabled = false;
                        this.textContent = '获取验证码';
                        this.style.opacity = '1';
                        this.style.cursor = 'pointer';
                    }
                }, 1000);
            });

            // 4. 表单提交阻止默认行为 (仅演示)
            document.querySelectorAll('form').forEach(form => {
                form.addEventListener('submit', (e) => {
                    e.preventDefault();
                    const btn = form.querySelector('.btn-submit');
                    const originalText = btn.textContent;
                    btn.textContent = '验证中...';
                    btn.style.opacity = '0.7';
                    
                    setTimeout(() => {
                        btn.textContent = '登录成功';
                        btn.style.background = '#10b981'; // 绿色
                        setTimeout(() => {
                            alert('欢迎进入舆情监控系统');
                            btn.textContent = originalText;
                            btn.style.background = '';
                            btn.style.opacity = '1';
                        }, 500);
                    }, 1500);
                });
            });
        });
    </script>
